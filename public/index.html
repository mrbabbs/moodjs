MoodJS test page


Open the console and type commands to try the library. Some examples: 

      
        var happyFaces = MoodJS.get('happy'); // Array that contains list of happy faces
        var happyOne = happyFaces[0];

        happyOne.setPercentageWithAnimation(50); // set value with animation
        happyOne.setPercentage(75); // set value withou animation

        happyOne.hideNose(); // hide nose

        // change color and show the nose
        happyOne
        .changeColorNose('#001122')
        .show();

        var newFace = MoodJS.add('sad', '#newFace')
          .setPercentageWithAnimation(35)
          .changeColorNose('#3476a3');
